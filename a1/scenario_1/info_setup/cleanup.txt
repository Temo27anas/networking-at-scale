# Resource cleanups
gcloud compute instances delete vm-finland vm-frankfurt vm-madrid --quiet
gcloud pubsub subscriptions delete sub-finland sub-frankfurt sub-madrid
gcloud pubsub topics delete fanout-topic



# Transfer scripts to VMs from local machine
vms=("vm-finland:europe-north1-a" "vm-frankfurt:europe-west3-a" "vm-madrid:europe-southwest1-a")

# Loop to copy both scripts to every VM
for entry in "${vms[@]}"; do
    vm="${entry%%:*}"
    zone="${entry#*:}"
    echo "Uploading to $vm..."
    gcloud compute scp publisher.py subscriber.py $vm:~/ --zone=$zone
done



# Setup venv and pip packages in VMs
sudo apt-get update
sudo apt-get install -y python3-venv
python3 -m venv ~/pubsub-env
~/pubsub-env/bin/pip install google-cloud-pubsub numpy



# Run in all 3 VMs (venv)
~/pubsub-env/bin/python3 subscriber.py --project_id=scale315 --sub_id=sub-finland
~/pubsub-env/bin/python3 subscriber.py --project_id=scale315 --sub_id=sub-frankfurt
~/pubsub-env/bin/python3 subscriber.py --project_id=scale315 --sub_id=sub-madrid

(on publisher VM, in this case Finland)
~/pubsub-env/bin/python3 publisher.py --project_id=scale315
